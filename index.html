import * as THREE from 'three'; // --- CONFIG --- const MAP_SIZE = 50; const WALL_THICKNESS = 1.5; let scene, camera, renderer, clock, player, flashLight, audioCtx; let yaw = new THREE.Object3D(), pitch = new THREE.Object3D(); let keys = {}, items = [], walls = []; let battery = 100, stamina = 100, dataFound = 0, isDead = false; let enemy, enemyActive = false; // --- INITIALIZE --- window.startGame = () => { document.getElementById('menu').style.display = 'none'; document.body.requestPointerLock(); if(!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)(); init(); }; function init() { scene = new THREE.Scene(); scene.fog = new THREE.Fog(0x000000, 1, 15); camera = new THREE.PerspectiveCamera(75, window.innerWidth/window.innerHeight, 0.1, 1000); player = new THREE.Group(); player.position.set(0, 1.6, 20); scene.add(player); player.add(yaw); yaw.add(pitch); pitch.add(camera); // Spotlight with Shadow flashLight = new THREE.SpotLight(0xffffff, 0, 25, Math.PI/5, 0.5, 1); flashLight.castShadow = true; pitch.add(flashLight); flashLight.target.position.set(0, 0, -1); pitch.add(flashLight.target); renderer = new THREE.WebGLRenderer({ antialias: true }); renderer.shadowMap.enabled = true; renderer.setSize(window.innerWidth, window.innerHeight); document.body.appendChild(renderer.domElement); clock = new THREE.Clock(); buildWorld(); spawnStalker(); spawnData(); document.addEventListener('mousemove', (e) => { if (document.pointerLockElement === document.body) { yaw.rotation.y -= e.movementX * 0.002; pitch.rotation.x -= e.movementY * 0.002; pitch.rotation.x = Math.max(-1.5, Math.min(1.5, pitch.rotation.x)); } }); animate(); } function buildWorld() { // Floor const floor = new THREE.Mesh(new THREE.PlaneGeometry(100, 100), new THREE.MeshStandardMaterial({color: 0x050505})); floor.rotation.x = -Math.PI/2; floor.receiveShadow = true; scene.add(floor); // Maze with Collision Data const wallMat = new THREE.MeshStandardMaterial({ color: 0x111111 }); const wallGeo = new THREE.BoxGeometry(4, 5, 4); for(let i=0; i<40; i++) { const wall = new THREE.Mesh(wallGeo, wallMat); wall.position.set(Math.floor(Math.random()*40-20), 2.5, Math.floor(Math.random()*40-20)); wall.castShadow = true; wall.receiveShadow = true; scene.add(wall); // Box Helper untuk deteksi tabrakan walls.push(new THREE.Box3().setFromObject(wall)); } } function spawnData() { const geo = new THREE.SphereGeometry(0.3, 16, 16); const mat = new THREE.MeshBasicMaterial({ color: 0x00d4ff }); for(let i=0; i<5; i++) { const p = new THREE.Mesh(geo, mat); p.position.set(Math.random()*36-18, 1, Math.random()*36-18); scene.add(p); items.push(p); } } function spawnStalker() { enemy = new THREE.Mesh(new THREE.BoxGeometry(0.8, 3, 0.8), new THREE.MeshBasicMaterial({color: 0x000000})); enemy.position.set(-20, 1.5, -20); scene.add(enemy); setTimeout(() => enemyActive = true, 5000); } // --- PHYSICS ENGINE (FIX TEMBUS TEMBOK) --- function checkCollision(newPos) { const playerBox = new THREE.Box3().setFromCenterAndSize(newPos, new THREE.Vector3(0.8, 1.8, 0.8)); for(let wall of walls) { if(playerBox.intersectsBox(wall)) return true; } return false; } function update() { if(isDead) return; const delta = clock.getDelta(); // Movement Logic const speed = (keys['ShiftLeft'] && stamina > 0) ? 0.2 : 0.08; const move = new THREE.Vector3(); if(keys['KeyW']) move.z -= 1; if(keys['KeyS']) move.z += 1; if(keys['KeyA']) move.x -= 1; if(keys['KeyD']) move.x += 1; if(move.length() > 0) { move.normalize().applyQuaternion(yaw.quaternion).multiplyScalar(speed); const nextPos = player.position.clone().add(move); if(!checkCollision(nextPos)) { player.position.copy(nextPos); if(speed > 0.1) stamina -= 0.5; } } stamina = Math.min(100, stamina + 0.1); document.getElementById('stam-fill').style.width = stamina + "%"; // Item Pickup items.forEach((it, i) => { if(player.position.distanceTo(it.position) < 1.5) { scene.remove(it); items.splice(i, 1); dataFound++; document.getElementById('dc').innerText = dataFound; if(dataFound === 5) { alert("SQUAD WIN!"); location.reload(); } } }); // Flashlight 'E' Toggle logic if(keys['KeyE']) { delete keys['KeyE']; // Prevent rapid toggle const on = flashLight.intensity > 0; flashLight.intensity = on ? 0 : 15; } // Enemy AI if(enemyActive) { const dist = player.position.distanceTo(enemy.position); const dir = new THREE.Vector3().subVectors(player.position, enemy.position).normalize(); enemy.position.add(dir.multiplyScalar(0.04)); document.getElementById('bpm').innerText = Math.floor(70 + (10/dist)*30); if(dist < 1.3) { isDead = true; document.getElementById('jumpscare').style.display='flex'; setTimeout(()=>location.reload(), 3000); } } } function animate() { if(!isDead) requestAnimationFrame(animate); update(); renderer.render(scene, camera); } window.onkeydown = (e) => keys[e.code] = true; window.onkeyup = (e) => keys[e.code] = false;
